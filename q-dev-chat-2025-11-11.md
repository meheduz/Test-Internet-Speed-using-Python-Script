CREATE OR REPLACE FUNCTION add_speed_label_column()
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 
    FROM information_schema.columns 
    WHERE table_name = 'speed_log' 
    AND column_name = 'speed_label'
  ) THEN
    ALTER TABLE speed_log ADD COLUMN speed_label TEXT;
    
    UPDATE speed_log 
    SET speed_label = CASE 
      WHEN download_mbps >= 100 THEN 'Excellent'
      WHEN download_mbps >= 50 THEN 'Great'
      WHEN download_mbps >= 25 THEN 'Good'
      WHEN download_mbps >= 10 THEN 'Fair'
      WHEN download_mbps >= 5 THEN 'Slow'
      ELSE 'Very Slow'
    END
    WHERE speed_label IS NULL;
  END IF;
END;
$$;
